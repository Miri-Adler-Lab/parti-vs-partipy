parti vs partipy

This notebook compares ParTI and ParTIpy archetypes for three datasets (HCM, HEP, and MSX). For each dataset and delta (0.0 and 0.5), it reads ARC files, computes Euclidean distances between archetypes, reorders matched pairs so the closest pairs are on the diagonal, draws a heatmap, and saves CSV/PNG outputs. At the end, it also saves one summary CSV and one Excel file with all tables.

The main inputs are ARC files that contain archetype locations in the first 3 PCs space. For ParTIpy, archetypes were generated using: pt.compute_archetypes(adata, n_archetypes, verbose=True, archetypes_only=False).

Then exported with: parti_arc = pd.DataFrame(adata.uns['AA_results']["Z"]). For ParTI, archetypes were generated with the ParTI run the run_parTI_sc.m script from the parTI_matlab_run folder.

import itertools

preferred_pc_cols = ["PC1", "PC2", "PC3"]

# Small config so you can switch behavior fast.
# `distance` means the matched pairs are ordered from closest to farthest on the diagonal.
calc_settings = {
    "hcm": {"use_sign_flip": False, "match_sort": "distance"},
    "hep": {"use_sign_flip": False, "match_sort": "distance"},
    "msx": {"use_sign_flip": False, "match_sort": "distance"},
}


def load_arc_pair(arc_folder: str, key: str) -> tuple[pd.DataFrame, pd.DataFrame]:
    parti_path = arc_root / arc_folder / f"{key}_parti.csv"
    partipy_path = arc_root / arc_folder / f"{key}_partipy.csv"

    if not parti_path.exists():
        raise FileNotFoundError(f"Missing file: {parti_path}")
    if not partipy_path.exists():
        raise FileNotFoundError(f"Missing file: {partipy_path}")

    return pd.read_csv(parti_path), pd.read_csv(partipy_path)


def pick_pc_cols(parti_df: pd.DataFrame, partipy_df: pd.DataFrame) -> list[str]:
    parti_cols = set(parti_df.columns)
    partipy_cols = set(partipy_df.columns)

    common = [c for c in preferred_pc_cols if c in parti_cols and c in partipy_cols]
    if not common:
        candidates = [c for c in (parti_cols & partipy_cols) if str(c).upper().startswith("PC")]
        common = sorted(candidates, key=lambda x: int(''.join(ch for ch in str(x) if ch.isdigit()) or 999))

    if len(common) < 2:
        raise ValueError("Need at least 2 shared PC columns.")

    return common


def _numeric_arc_arrays(parti_df: pd.DataFrame, partipy_df: pd.DataFrame) -> tuple[np.ndarray, np.ndarray, list[str], np.ndarray, np.ndarray]:
    if "archetype" not in parti_df.columns or "archetype" not in partipy_df.columns:
        raise ValueError("Both ARC files need an 'archetype' column.")

    pc_cols = pick_pc_cols(parti_df, partipy_df)

    a = parti_df[pc_cols].apply(pd.to_numeric, errors="coerce").to_numpy(dtype=float)
    b = partipy_df[pc_cols].apply(pd.to_numeric, errors="coerce").to_numpy(dtype=float)

    if np.isnan(a).any() or np.isnan(b).any():
        raise ValueError("Found non-numeric or NaN in PC columns.")

    row_labels = parti_df["archetype"].astype(str).to_numpy()
    col_labels = partipy_df["archetype"].astype(str).to_numpy()
    return a, b, pc_cols, row_labels, col_labels


def best_pc_sign_alignment(parti_arr: np.ndarray, partipy_arr: np.ndarray) -> tuple[np.ndarray, np.ndarray, float]:
    # Copied logic from old Xenium pipeline: try all +/- sign vectors, keep lowest Hungarian sum.
    dim = min(parti_arr.shape[1], partipy_arr.shape[1])
    if dim < 1:
        raise ValueError("No shared dimensions for sign alignment.")

    parti = parti_arr[:, :dim]
    partipy = partipy_arr[:, :dim]

    best_sign = np.ones(dim, dtype=float)
    best_cost = np.inf

    for signs in itertools.product([-1.0, 1.0], repeat=dim):
        sign_vec = np.asarray(signs, dtype=float)
        flipped = partipy * sign_vec
        d = cdist(parti, flipped, metric="euclidean")
        r, c = linear_sum_assignment(d)
        cost = float(d[r, c].sum())
        if cost < best_cost:
            best_cost = cost
            best_sign = sign_vec

    return parti, partipy * best_sign, float(best_cost), best_sign


def build_distance_df(parti_df: pd.DataFrame, partipy_df: pd.DataFrame, use_sign_flip: bool = False) -> pd.DataFrame:
    a, b, pc_cols, rows, cols = _numeric_arc_arrays(parti_df, partipy_df)

    sign_vec = np.ones(len(pc_cols), dtype=float)
    sign_cost = np.nan
    if use_sign_flip:
        a, b, sign_cost, sign_vec = best_pc_sign_alignment(a, b)

    d = cdist(a, b, metric="euclidean")
    out = pd.DataFrame(d, index=rows, columns=cols)
    out.attrs["pc_cols_used"] = pc_cols
    out.attrs["sign_vector"] = sign_vec.tolist()
    out.attrs["sign_cost"] = sign_cost
    return out


def reorder_to_diagonal_match(dist_df: pd.DataFrame, match_sort: str = "distance") -> tuple[pd.DataFrame, pd.DataFrame]:
    m = dist_df.to_numpy(dtype=float)
    row_idx, col_idx = linear_sum_assignment(m)

    pairs = list(zip(row_idx.tolist(), col_idx.tolist()))
    if match_sort == "row":
        pairs = sorted(pairs, key=lambda x: x[0])
    elif match_sort == "col":
        pairs = sorted(pairs, key=lambda x: x[1])
    elif match_sort == "distance":
        pairs = sorted(pairs, key=lambda x: m[x[0], x[1]])
    else:
        raise ValueError("match_sort must be 'row', 'col', or 'distance'.")

    rows = [r for r, _ in pairs]
    cols = [c for _, c in pairs]

    matched_df = dist_df.iloc[rows, cols].copy()
    matched_df.index = dist_df.index[rows]
    matched_df.columns = dist_df.columns[cols]

    matching_df = pd.DataFrame({
        "rank": np.arange(1, len(pairs) + 1, dtype=int),
        "ParTI": dist_df.index[rows].astype(str),
        "ParTIpy": dist_df.columns[cols].astype(str),
        "distance": m[rows, cols],
    })

    matched_df.attrs.update(dist_df.attrs)
    return matched_df, matching_df


def save_and_plot_dist(
    dist_df: pd.DataFrame,
    title: str,
    key: str,
    out_dir: Path,
) -> dict:
    out_dir.mkdir(parents=True, exist_ok=True)

    out_dist = out_dir / f"{key}_dist.csv"
    out_png = out_dir / f"{key}_panel.png"

    dist_df.to_csv(out_dist)

    M = dist_df.to_numpy(dtype=float)
    nrows, ncols = M.shape

    fig, ax = plt.subplots(figsize=(6, 5), dpi=150)
    im = ax.imshow(M, cmap="coolwarm", vmin=M.min(), vmax=M.max(), aspect="equal", interpolation="nearest")

    ax.set_xticks(range(ncols))
    ax.set_yticks(range(nrows))
    ax.set_xticklabels(range(1, ncols + 1))
    ax.set_yticklabels(range(1, nrows + 1))

    ax.set_xlabel("ParTIpy archetypes")
    ax.set_ylabel("ParTI archetypes")
    ax.set_title(title)

    for i in range(nrows):
        for j in range(ncols):
            ax.text(j, i, f"{M[i, j]:.2f}", ha="center", va="center", fontsize=8)

    cbar = fig.colorbar(im, ax=ax, fraction=0.046, pad=0.04)
    cbar.set_label("Euclidean Distance")

    for spine in ax.spines.values():
        spine.set_visible(False)

    plt.tight_layout()
    plt.savefig(out_png, bbox_inches="tight")
    plt.close(fig)

    return {
        "title": title,
        "key": key,
        "dist_df": dist_df,
        "dist_out": str(out_dist),
        "png": str(out_png),
    }


def run_dataset_case(dataset_id: str, arc_folder: str, key: str, title: str, out_folder: str) -> pd.DataFrame:
    cfg = calc_settings[dataset_id]

    parti_df, partipy_df = load_arc_pair(arc_folder=arc_folder, key=key)
    raw_df = build_distance_df(parti_df, partipy_df, use_sign_flip=cfg["use_sign_flip"])
    df, match_df = reorder_to_diagonal_match(raw_df, match_sort=cfg["match_sort"])

    r = save_and_plot_dist(
        dist_df=df,
        title=title,
        key=key,
        out_dir=out_root / out_folder,
    )

    match_out = out_root / out_folder / f"{key}_matching.csv"
    match_df.to_csv(match_out, index=False)

    r["matching_df"] = match_df
    r["match_out"] = str(match_out)
    r["pc_cols_used"] = raw_df.attrs.get("pc_cols_used", [])
    r["sign_vector"] = ",".join(str(int(v)) for v in raw_df.attrs.get("sign_vector", []))
    r["sign_cost"] = raw_df.attrs.get("sign_cost", np.nan)
    r["use_sign_flip"] = bool(cfg["use_sign_flip"])
    r["match_sort"] = cfg["match_sort"]

    all_results.append(r)
    return df

# final export: one workbook with summary + all dist + matching + ARC sheets
excel_path = out_root / "all_distance_tables.xlsx"


def arc_folder_for_key(key: str) -> str:
    if key.startswith("hcm_"):
        return "hcm"
    if key.startswith("hep_"):
        return "hep"
    if key.startswith("msx_"):
        return "msx"
    raise ValueError(f"Unknown key prefix for ARC lookup: {key}")


arc_files_root = repo / "arc_files"

with pd.ExcelWriter(excel_path) as writer:
    summary_df.to_excel(writer, sheet_name="summary", index=False)

    for r in all_results:
        key = r["key"]

        # distance sheet (already diagonal-matched order)
        dist_sheet = key[:31]
        r["dist_df"].to_excel(writer, sheet_name=dist_sheet)

        # matching sheet
        if "matching_df" in r:
            match_sheet = f"{key}_match"[:31]
            r["matching_df"].to_excel(writer, sheet_name=match_sheet, index=False)

        # arc sheets (ParTI + ParTIpy)
        arc_folder = arc_folder_for_key(key)
        parti_path = arc_files_root / arc_folder / f"{key}_parti.csv"
        partipy_path = arc_files_root / arc_folder / f"{key}_partipy.csv"

        if parti_path.exists():
            parti_df = pd.read_csv(parti_path)
            parti_sheet = f"{key}_parti"[:31]
            parti_df.to_excel(writer, sheet_name=parti_sheet, index=False)

        if partipy_path.exists():
            partipy_df = pd.read_csv(partipy_path)
            partipy_sheet = f"{key}_partipy"[:31]
            partipy_df.to_excel(writer, sheet_name=partipy_sheet, index=False)

print("Excel saved:", excel_path)
